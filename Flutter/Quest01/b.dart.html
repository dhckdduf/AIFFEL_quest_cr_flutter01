import 'dart:async';

class PomodoroTimer {
  int workDuration = 5; // 25초 작업
  int shortBreakDuration = 1; // 5초 휴식
  int longBreakDuration = 3; // 15초 긴 휴식 (사용하지 않음)
  int cycleCount = 0; // 현재 사이클(작업+휴식) 수
  Timer? _timer;
  int remainingTime = 0;

  void startPomodoro() {
    print("flutter: Pomodoro 타이머를 시작합니다.");
    cycleCount = 0;
    _startWork();
  }

  void _startWork() {
    remainingTime = workDuration;
    _startCountdown((){
      print("flutter: 작업 시간이 종료되었습니다. 휴식 시간을 시작합니다.");
      _startBreak();
    });
  }

  void _startBreak() {
    cycleCount++;
    if (cycleCount % 4 ==0){
      remainingTime = longBreakDuration;
      print("flutter: 4회차! 긴 휴식을 시작합니다.");
    } else {
      remainingTime = shortBreakDuration;
    }
    _startCountdown((){
      print("flutter: 휴식 시간이 종료되었습니다. 새로운 작업을 시작합니다.");
      _startWork();
    });
  }

  void _startCountdown(Function onComplete) {
    _timer = Timer.periodic(Duration(seconds: 1), (timer) {
      if (remainingTime <= 0) {
        timer.cancel();
        onComplete();
      } else {
        print("flutter: ${_formatTime(remainingTime)}");
        remainingTime--;
      }
    });
  }

  String _formatTime(int seconds) {
    int minutes = seconds ~/ 60;
    int sec = seconds % 60;
    return '$minutes:${sec.toString().padLeft(2, '0')}';
  }
}

void main() {
  PomodoroTimer pomodoro = PomodoroTimer();
  pomodoro.startPomodoro();
}
